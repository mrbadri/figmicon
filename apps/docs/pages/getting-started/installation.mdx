# Installation

## Using package manager

```bash
# npm
npm install @iconsync/core

# yarn
yarn add @iconsync/core

# pnpm
pnpm install @iconsync/core
```

## Basic Usage

```jsx
import { Button } from '@iconsync/core';

export default function App() {
  return (
    <Button variant="primary">
      Click me
    </Button>
  );
}
```
